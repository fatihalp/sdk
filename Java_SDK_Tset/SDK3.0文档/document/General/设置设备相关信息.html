设置设备相关信息

SetDeviceName设置设备名

<!--请求
##value 设备名
-->
<?xml version='1.0' encoding='utf-8'?>
<sdk guid="##GUID">
    <in method="SetDeviceName">
        <name value="##value"/>
    </in>
</sdk>

<!--反馈-->
<?xml version="1.0" encoding="utf-8"?>
<sdk guid="fecc9420c96a6a708b35e13f62385e15">
    <out method="SetDeviceName" result="kSuccess"/>
</sdk>
